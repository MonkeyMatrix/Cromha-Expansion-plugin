# Fuel Exportation Jobs
mission empire-fuel-transport[0]
	job
	repeat
	name "Fuel Export to <planet>"
	description "Transport 2 Fuel Capacitors full of non-refined oil to <destination> before <date>. Payment is <payment>."
	deadline 4 2
	destination
		not government "Empire Rebels" "Kor Mereti" "Kor Efret" "Kor Sestor" "Hai (Unfettered)" "Pirate"
		distance 2 4
	source
		government "Cromha Empire" "Mineria Colonies"
	to offer
		random < 5
		"reputation: Cromha Empire" > 100
	on accept
		outfit "Fuel Capacitor" +2
		dialog "A member of the Mineria Colonies personnel gives you the 2 Fuel Capacitor and leaves your ship."
	on complete
		outfit "Fuel Capacitor" -2
		"reputation: Cromha Empire" += 10
		"reputation: Mineria Colonies" += 20
		dialog "You give the Fuel Capacitors to a Mineria colony personnel and receive <payment>."
		payment
		payment 50000 3

# Secure deliveries
mission secure-deliver[0]
	name "Secure Deliver to <planet>"
	description "Deliver a secure package to <destination> before <date>. Payment is <payment>."
	deadline 2 9
	job
	repeat
	source
		government "Cromha Empire" "Mineria Colonies"
	destination
		distance 2 12
		attributes military
		government "Cromha Empire" "Mineria Colonies" "Hai" "Quarg" "Syndicate" "Remnant" "Wanderer"
	cargo "secure package" 1
	to offer
		random < 25
		"reputation: Cromha Empire" >= 35
		has not "secure empire package ability"
	on accept
		dialog phrase "secure package pickup"
	on visit
		dialog phrase "generic cargo on visit"
	on abort
		dialog "The Cromha Empire will not be happy. You were fined 250,000 credits."
		set "secure empire package ability"
		fine 250000
	on fail
		dialog "The Cromha Empire will not be happy. You were fined 250,000 credits."
		set "secure empire package ability"
		fine 250000
	on complete
		payment 70000 750
		"reputation: Cromha Empire" += 1
		"secure empire package" ++
		dialog phrase "secure package dropoff payment"
